\begin{algorithmic}[1]
    \Function {Semivariance}{$y, \mu$}
        \State $N \gets sizeof(y)$
        \State $z \gets 0$
        \For {$i \gets 1, n$}
            \If {$y_i > \mu$}
                \State $z \gets z + (y_i - \mu)^2$
            \EndIf
        \EndFor
        \State $z \gets z / (N-1)$
        \State \Return $z$
    \EndFunction
\end{algorithmic}
